Update Dropdown options via state after initial render:

    initialState = {items: ["Velociraptor", "Tyrannosaurus"]}
    rows = state.items.map((item, index) => (
      <li value={item} key="{index}">
        {item}
      </li>
    ));
    handleToggle = () => {
        setState({ items: ["Diplodocus", "Triceratops", "Brontosaurus"] })
    };
    <div> 
      <Dropdown defaultText="Select One ..." label="Update State Dropdown" onChange={(value, event) => {console.log("onChange: ", value, event)}} >
        {rows}
        </Dropdown><br />
        <Button primary onClick={handleToggle}>Update State</Button>
        <br /><br />
        List: {state.items.toString()}
    </div>
